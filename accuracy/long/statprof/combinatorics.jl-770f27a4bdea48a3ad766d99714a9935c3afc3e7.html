161 """
162 permute!(v, p)
163
164 Permute vector `v` in-place, according to permutation `p`. No checking is done
165 to verify that `p` is a permutation.
166
167 To return a new permutation, use `v[p]`. This is generally faster than `permute!(v, p)`;
168 it is even faster to write into a pre-allocated output array with `u .= @view v[p]`.
169 (Even though `permute!` overwrites `v` in-place, it internally requires some allocation
170 to keep track of which elements have been moved.)
171
172 $(_DOCS_ALIASING_WARNING)
173
174 See also [`invpermute!`](@ref).
175
176 # Examples
177 ```jldoctest
178 julia> A = [1, 1, 3, 4];
179
180 julia> perm = [2, 4, 3, 1];
181
182 julia> permute!(A, perm);
183
184 julia> A
185 4-element Vector{Int64}:
186 1
187 4
188 3
189 1
190 ```

288 function invperm(a::AbstractVector)
289 require_one_based_indexing(a)
290 b = zero(a) # similar vector of zeros
291 n = length(a)
292 @inbounds for (i, j) in enumerate(a)
293 ((1 <= j <= n) && b[j] == 0) ||
294 throw(ArgumentError("argument is not a permutation"))
295 b[j] = i
296 end
297 b
298 end

324 Next integer greater than or equal to `n` that can be written as ``\\prod k_i^{p_i}`` for integers

339 function nextprod(a::Union{Tuple{Vararg{Integer}},AbstractVector{<:Integer}}, x::Real)
340 if x > typemax(Int)
341 throw(ArgumentError("unsafe for x > typemax(Int), got $x"))
342 end
343 k = length(a)
344 v = fill(1, k) # current value of each counter
345 mx = map(a -> nextpow(a,x), a) # maximum value of each counter
346 v[1] = mx[1] # start at first case that is >= x
347 p::widen(Int) = mx[1] # initial value of product in this case
348 best = p
349 icarry = 1
350
351 while v[end] < mx[end]
352 if p >= x
353 best = p < best ? p : best # keep the best found yet
354 carrytest = true
355 while carrytest
356 p = div(p, v[icarry])
357 v[icarry] = 1
358 icarry += 1
359 p *= a[icarry]
360 v[icarry] *= a[icarry]
361 carrytest = v[icarry] > mx[icarry] && icarry < k
362 end
363 if p < x
364 icarry = 1
365 end
366 else
367 while p < x
368 p *= a[1]
369 v[1] *= a[1]
370 end
371 end
372 end
373 # might overflow, but want predictable return type
374 return mx[end] < best ? Int(mx[end]) : Int(best)
